3   import java.awt.Dimension;
4   import java.awt.Font;
5   import javax.swing.JButton;
6   import javax.swing.JComboBox;
7   import javax.swing.JLabel;
8   import javax.swing.JPanel;
9   import ui.listeners.MenuListener;
10  
11  /**
12   * This class is used to create the main menu.
13   *
14   * @author eerop
15   */
16  public class Menu extends JPanel {
17  
18      private JButton simulate, speedtest, resize;
19      private JComboBox<String> dropdown;
20      private Font font;
21      private MenuListener ml;
22  
23      /**
24       * Constructor for this class.
25       *
26       * @param ml Menulistener.
27       */
28      public Menu(MenuListener ml) {
29          this.ml = ml;
30          setLayout(null);
31          setPreferredSize(new Dimension(256, 640));
32          font = new Font("Verdana", Font.BOLD, 20);
33          setButtons();
34          setDropdown();
35          ml.setButtons(simulate, speedtest, resize);
36          ml.setDropdown(dropdown);
37          simulate.addActionListener(ml);
38          speedtest.addActionListener(ml);
39          resize.addActionListener(ml);
40  
41          Font h = new Font("Verdana", Font.BOLD, 30);
42          JLabel header1 = new JLabel("Algorithm-");
43          header1.setFont(h);
44          header1.setBounds(36, 32, 196, 64);
45          add(header1);
46          JLabel header2 = new JLabel("Analyser");
47          header2.setFont(h);
48          header2.setBounds(52, 70, 196, 64);
49          add(header2);
50      }
51  
52      private void setButtons() {
53          this.simulate = new JButton("Simulate");
54          simulate.setBounds(32, 192, 192, 32);
55          simulate.setFont(font);
56          this.speedtest = new JButton("Speedtest");
57          speedtest.setBounds(32, 256, 192, 32);
58          speedtest.setFont(font);
59          add(simulate);
60          add(speedtest);
61      }
62  
63      private void setDropdown() {
64          JLabel label = new JLabel("Set size:");
65          label.setBounds(32, 320, 192, 32);
66          label.setFont(font);
67          add(label);
68          this.dropdown = new JComboBox(choises());
69          dropdown.setVisible(true);
70          dropdown.setBounds(32, 352, 192, 32);
71          dropdown.setFont(font);
72          dropdown.setSelectedIndex(5);
73          this.resize = new JButton("Resize");
74          resize.setBounds(64, 416, 128, 32);
75          resize.setFont(font);
76          add(dropdown);
77          add(resize);
78      }
79  
80      private String[] choises() {
81          String[] r = new String[26];
82          for (int i = 0; i < 26; i++) {
83              r[i] = "" + (15 + (i * 3));
84          }
85          return r;
86      }
87  }

1   package ui.testresultwindow;
2   
3   import java.awt.Dimension;
4   import java.awt.Font;
5   import javax.swing.JButton;
6   import javax.swing.JCheckBox;
7   import javax.swing.JLabel;
8   import javax.swing.JPanel;
9   
10  /**
11   * Creates the window where user can choose what algorithms are going to be
12   * included in the comparison.
13   *
14   * @author eerop
15   */
16  public class AlgorithmSelection extends JPanel {
17  
18      private JCheckBox aManhattan, aEuclidean, aOctile, aChebyshev, bManhattan, bEuclidean,
19              bOctile, bChebyshev, breadth, depth, dijkstra;
20      private JButton ready;
21      private Font heuristics, algorithms;
22      private TestRunner tr;
23  
24      /**
25       * Basic constructor.
26       *
27       * @param tr testrunner
28       */
29      public AlgorithmSelection(TestRunner tr) {
30          this.setLayout(null);
31          this.setPreferredSize(new Dimension(960, 544));
32          setFonts();
33          initializeComponents();
34          setPositionings();
35          addComponents();
36          addLabels();
37  
38          this.tr = tr;
39          tr.setComponents(aManhattan, aEuclidean, aOctile, aChebyshev, bManhattan, bEuclidean,
40                  bOctile, bChebyshev, breadth, depth, dijkstra, ready);
41          ready.addActionListener(tr);
42      }
43  
44      private void initializeComponents() {
45          this.aManhattan = new JCheckBox("Manhattan");
46          aManhattan.setFont(heuristics);
47          this.aEuclidean = new JCheckBox("Euclidean");
48          aEuclidean.setFont(heuristics);
49          this.aOctile = new JCheckBox("Octile");
50          aOctile.setFont(heuristics);
51          this.aChebyshev = new JCheckBox("Chebyshev");
52          aChebyshev.setFont(heuristics);
53          this.bManhattan = new JCheckBox("Manhattan");
54          bManhattan.setFont(heuristics);
55          this.bEuclidean = new JCheckBox("Euclidean");
56          bEuclidean.setFont(heuristics);
57          this.bOctile = new JCheckBox("Octile");
58          bOctile.setFont(heuristics);
59          this.bChebyshev = new JCheckBox("Chebyshev");
60          bChebyshev.setFont(heuristics);
61          this.breadth = new JCheckBox("BFS");
62          breadth.setFont(algorithms);
63          this.depth = new JCheckBox("DFS");
64          depth.setFont(algorithms);
65          this.dijkstra = new JCheckBox("Dijkstra");
66          dijkstra.setFont(algorithms);
67          this.ready = new JButton("Continue");
68          ready.setFont(algorithms);
69      }
70  
71      private void setPositionings() {
72          aManhattan.setBounds(64, 224, 160, 32);
73          aEuclidean.setBounds(64, 256, 160, 32);
74          aOctile.setBounds(64, 288, 160, 32);
75          aChebyshev.setBounds(64, 320, 160, 32);
76          bManhattan.setBounds(256, 224, 160, 32);
77          bEuclidean.setBounds(256, 256, 160, 32);
78          bOctile.setBounds(256, 288, 160, 32);
79          bChebyshev.setBounds(256, 320, 160, 32);
80          breadth.setBounds(448, 192, 160, 32);
81          depth.setBounds(608, 192, 160, 32);
82          dijkstra.setBounds(768, 192, 160, 32);
83          ready.setBounds(320, 448, 320, 64);
84      }
85  
86      private void addComponents() {
87          add(aManhattan);
88          add(aEuclidean);
89          add(aOctile);
90          add(aChebyshev);
91          add(bManhattan);
92          add(bEuclidean);
93          add(bOctile);
94          add(bChebyshev);
95          add(breadth);
96          add(depth);
97          add(dijkstra);
98          add(ready);
99      }
100 
101     private void addLabels() {
102         JLabel a = new JLabel("A*:");
103         a.setFont(algorithms);
104         a.setBounds(32, 192, 160, 32);
105         add(a);
106         JLabel best = new JLabel("Best-first:");
107         best.setFont(algorithms);
108         best.setBounds(224, 192, 160, 32);
109         add(best);
110         JLabel select = new JLabel("Select algorithms to be tested:");
111         select.setFont(algorithms);
112         select.setBounds(256, 32, 640, 64);
113         add(select);
114     }
115 
116     private void setFonts() {
117         this.heuristics = new Font("Verdana", Font.BOLD, 18);
118         this.algorithms = new Font("Verdana", Font.BOLD, 25);
119     }
120 }

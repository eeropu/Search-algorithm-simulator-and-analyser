37      /**
38       * Constructor for this class.
39       *
40       * @param wh reference to the windowhandler
41       * @param grid reference to the grid
42       */
43      public SubMenu(WindowHandler wh, Grid grid) {
44          this.font = new Font("Verdana", Font.BOLD, 18);
45          setLayout(null);
46          setPreferredSize(new Dimension(256, 640));
47          setComponents();
48          this.sml = new SubMenuListener(jcb, addWeight, setHeuristic, start, goal, wall, weightButton, manhattan, euclidean, octile, chebyshev, yes, no, activate, clear, fill, back, help, weightText);
49          this.stwo = new SpeedTestWindowOpener(activate, grid, yes);
50          sml.setGrid(grid);
51          sml.setWH(wh);
52          setListener();
53          setSpeed();
54      }
55  
56      private void setComponents() {
57          jcb = new JComboBox<>(algorithms());
58          jcb.setBounds(32, 32, 192, 32);
59          add(jcb);
60          setLabels();
61          setRadioButtons();
62          weightSetter();
63          selectHeuristic();
64          allowDiagonal();
65          setButtons();
66      }
67  
68      private String[] algorithms() {
69          String[] r = new String[5];
70          r[0] = "A*";
71          r[1] = "Best-first-search";
72          r[2] = "Breadth-first-search";
73          r[3] = "Depth-first-search";
74          r[4] = "Dijkstra";
75          return r;
76      }

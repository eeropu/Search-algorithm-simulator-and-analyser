17  
18      public WindowHandler() {
19          cl = new CardLayout();
20          cardPanel = new JPanel(cl);
21          cardPanel.setBounds(960, 0, 160, 640);
22          panel = new JPanel();
23          panel.setLayout(null);
24          grid = new Grid();
25          panel.add(grid);
26          panel.add(cardPanel);
27      }
28  
29      @Override
30      public void run() {
31          JFrame frame = new JFrame("Algorithm analyser");
32          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
33          frame.setVisible(true);
34          frame.getContentPane().setPreferredSize(new Dimension(1216, 640));
35          frame.setResizable(false);
36          frame.add(panel);
37          frame.pack();
38      }

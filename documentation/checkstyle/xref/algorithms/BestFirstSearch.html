6   /**
7    * This class is used to run a Best-First-Search.
8    *
9    * @author eerop
10   */
11  public class BestFirstSearch extends AlgorithmBase {
12  
13      private final Heap h;
14      private boolean ready;
15  
16      /**
17       * Constructor for this class.
18       *
19       * @param current the start-vertex of the given Grid.
20       */
21      public BestFirstSearch(Vertex current) {
22          h = new Heap();
23          h.insert(current);
24          this.current = current;
25          this.ready = false;
26      }
27  
28      @Override
29      public void run() {
30          Vertex a = current.getNeighbour();
31          if (a == null) {
32              return;
33          }
34          if (a.getMode() == 'w') {
35              h.insert(a);
36              a.setMode('g');
37              a.refresh();
38              a.setPrev(current);
39          } else if (a.getMode() == 'f') {
40              a.setPrev(current);
41              this.f = a;
42              this.ready = true;
43          }
44      }
45  
46      @Override
47      public void next() {
48          current.setMode('d');
49          current.refresh();
50          current = h.delMin();
51      }
52  
53      @Override
54      public boolean done() {
55          return h.isEmpty() || ready;
56      }
57  
58  }
